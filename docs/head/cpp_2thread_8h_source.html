375 template <class Function, class... Args>
376 Error Thread::Start(Function&& func, Args&&... args)
377 {
378  if (thread_)
379  return kEtcPalErrInvalid;
380 
381  thread_.reset(new etcpal_thread_t);
382 
383  auto new_f = std::unique_ptr<FunctionType>(
384  new FunctionType(std::bind(std::forward<Function>(func), std::forward<Args>(args)...)));
385  Error create_res = etcpal_thread_create(thread_.get(), &params_, CppThreadFn, new_f.get());
386  if (create_res)
387  new_f.release();
388  else
389  thread_.reset();
390  return create_res;
391 }

424 template <typename Rep, typename Period>
425 void Thread::Sleep(const std::chrono::duration<Rep, Period>& sleep_duration) noexcept
426 {
427  // This implementation cannot sleep longer than UINT_MAX.
428  unsigned int sleep_ms_clamped = static_cast<unsigned int>(
429  std::min(std::chrono::milliseconds(sleep_duration).count(),
430  static_cast<std::chrono::milliseconds::rep>(std::numeric_limits<unsigned int>::max())));
431  Sleep(sleep_ms_clamped);
432 }
