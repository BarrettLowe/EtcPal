300 // The ExpectedStorage class provides a way to conditionally delete the copy and/or move
301 // constructors of etcpal::Expected based on the copyability and moveability of the T that we are
302 // being specialized with.
303 //
304 // We provide a generic template where the copy and move constructors are both deleted, then 3
305 // specializations, for (copyable and movable), (copyable and not moveable), and (moveable and not
306 // copyable) respectively.
307 template <typename T, bool IsCopyConstructible, bool IsMoveConstructible>
308 class ExpectedStorage : public ExpectedStorageImpl<T>
309 {
310 public:
311  ExpectedStorage() = default;
312  ~ExpectedStorage() = default;
313 
314  explicit ExpectedStorage(bool has_value) : ExpectedStorageImpl<T>(has_value) {}
315 
316  ExpectedStorage(const ExpectedStorage& other) = delete;
317  ExpectedStorage(ExpectedStorage&& other) = delete;
318 };
319 
320 template <typename T>
321 class ExpectedStorage<T, true, true> : public ExpectedStorageImpl<T>
322 {
323 public:
324  ExpectedStorage() = default;
325  ~ExpectedStorage() = default;
326 
327  explicit ExpectedStorage(bool has_value) : ExpectedStorageImpl<T>(has_value) {}
328 
329  ExpectedStorage(const ExpectedStorage& other) : ExpectedStorageImpl<T>(other.has_value())
330  {
331  if (this->has_value())
332  this->ConstructValue(other.value());
