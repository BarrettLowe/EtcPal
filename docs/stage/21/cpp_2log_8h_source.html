263 inline bool Logger::Startup(LogMessageHandler& message_handler)
264 {
265  if (etcpal_init(ETCPAL_FEATURE_LOGGING) != kEtcPalErrOk)
266  return false;
267 
268  if (!etcpal_validate_log_params(&log_params_))
269  {
270  etcpal_deinit(ETCPAL_FEATURE_LOGGING);
271  return false;
272  }
273 
274  log_params_.context = &message_handler;
275 
276  if (dispatch_policy_ == LogDispatchPolicy::Queued)
277  {
278  // Start the log dispatch thread
279  running_ = true;
280  signal_.reset(new etcpal::Signal);
281  mutex_.reset(new etcpal::Mutex);
282  if (thread_.SetName("EtcPalLoggerThread").Start(&Logger::LogThreadRun, this))
283  {
284  return true;
285  }
286  else
287  {
288  etcpal_deinit(ETCPAL_FEATURE_LOGGING);
289  return false;
290  }
291  }
292  else
293  {
294  running_ = true;
295  return true;
296  }
297 }

557 inline void Logger::LogInternal(int pri, const char* format, std::va_list args)
558 {
559  if (running_)
560  {
561  if (dispatch_policy_ == LogDispatchPolicy::Direct)
562  {
563  etcpal_vlog(&log_params_, pri, format, args);
564  }
565  else
566  {
567  {
568  etcpal::MutexGuard lock(*mutex_);
569  msg_q_.emplace();
570  msg_q_.back().pri = pri;
571  vsnprintf(msg_q_.back().buf, ETCPAL_RAW_LOG_MSG_MAX_LEN, format, args);
572  }
573  signal_->Notify();
574  }
575  }
576 }

578 inline void Logger::LogThreadRun()
579 {
580  while (running_)
581  {
582  EmptyLogQueue();
583  signal_->Wait();
584  }
585  // Bail the queue one last time on exit
586  EmptyLogQueue();
587 }
588 
589 inline void Logger::EmptyLogQueue()
590 {
591  std::queue<LogMessage> to_log;
592  {
593  etcpal::MutexGuard lock(*mutex_);
594  to_log.swap(msg_q_);
595  }
596 
597  while (!to_log.empty())
598  {
599  etcpal_log(&log_params_, to_log.front().pri, to_log.front().buf);
600  to_log.pop();
601  }
602 }
