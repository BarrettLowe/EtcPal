196 /* Timestamp length:
197  * Date: 10 (1970-01-01)
198  * "T" or " ": 1
199  * Time: 12 (00:00:00.001)
200  * UTC Offset: 6 (-05:00)
201  * Nullterm: 1
202  * --------------
203  * Total: 30
204  */
206 #define ETCPAL_LOG_TIMESTAMP_LEN 30u
207 
208 /* Syslog header length (from RFC 5424):
209  * PRIVAL: 5
210  * VERSION: 3
211  * SP: 1
212  * TIMESTAMP: [Referenced]
213  * SP: 1
214  * HOSTNAME: [Referenced]
215  * SP: 1
216  * APP-NAME: [Referenced]
217  * SP: 1
218  * PROCID: [Referenced]
219  * SP: 1
220  * MSGID (not used): 1
221  * SP: 1
222  * STRUCTURED-DATA (not used): 1
223  * SP: 1
224  * -----------------------------
225  * Total non-referenced: 17
226  */
228 #define ETCPAL_SYSLOG_HEADER_MAX_LEN \
229  (17u + (ETCPAL_LOG_TIMESTAMP_LEN - 1u) + (ETCPAL_LOG_HOSTNAME_MAX_LEN - 1u) + (ETCPAL_LOG_APP_NAME_MAX_LEN - 1u) + \
230  (ETCPAL_LOG_PROCID_MAX_LEN - 1u))
231
