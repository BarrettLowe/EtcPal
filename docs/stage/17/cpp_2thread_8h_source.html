369 template <class Function, class... Args>
370 Result Thread::Start(Function&& func, Args&&... args)
371 {
372  if (thread_)
373  return kEtcPalErrInvalid;
374 
375  thread_.reset(new etcpal_thread_t);
376 
377  auto new_f = std::unique_ptr<FunctionType>(
378  new FunctionType(std::bind(std::forward<Function>(func), std::forward<Args>(args)...)));
379  Result create_res = etcpal_thread_create(thread_.get(), &params_, CppThreadFn, new_f.get());
380  if (create_res)
381  new_f.release();
382  else
383  thread_.reset();
384  return create_res;
385 }

418 template <typename Rep, typename Period>
419 void Thread::Sleep(const std::chrono::duration<Rep, Period>& sleep_duration) noexcept
420 {
421  // This implementation cannot sleep longer than UINT_MAX.
422  unsigned int sleep_ms_clamped = static_cast<unsigned int>(
423  std::min(std::chrono::milliseconds(sleep_duration).count(),
424  static_cast<std::chrono::milliseconds::rep>(std::numeric_limits<unsigned int>::max())));
425  Sleep(sleep_ms_clamped);
426 }
